Brackets 

        function check(str, bracketsConfig) {
            let bracketsPairs = bracketsConfig.reduce(
              (a, b) => ((a[b[0]] = b[1]), a),
              {}
            );
            let stack = [];
            for (let i = 0; i < str.length; i++) {
              const brack = str[i];
              if (brack == bracketsPairs[stack[stack.length - 1]]) {
                stack.pop();
              } else if (bracketsPairs[brack]) {
                stack.push(brack);
              } else {
                return false;
              }
            }
            return stack.length === 0;
          };
